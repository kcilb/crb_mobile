1. Overview

How to obtain a Bearer Token (JWT) for channel APIs. The token authenticates requests and is used to encrypt/decrypt payloads for the secured channel endpoints listed in Section 3 – Payload Encryption.

All requests/responses are JSON over HTTPS. Send the token in Authorization: Bearer <token> for all protected requests.

  • POST /api/v1/xapi-service/fundsTransfer
  • POST /api/v1/xapi-service/postBillPayment
  • POST /api/v1/xapi-service/processBillValidation

3.5 Code Examples

C# Encryption Example

using System;
using System.IO;
using System.Security.Cryptography;
using System.Text;
using System.Text.Json;
using System.IdentityModel.Tokens.Jwt;

public class EncryptionHelper
{
    private static readonly byte[] IV = Encoding.UTF8.GetBytes("1234567890123456");
    
    /// <summary>
    /// Extracts the AES key from a JWT bearer token
    /// </summary>
    public static byte[] ExtractAesKeyFromToken(string bearerToken)
    {
        var handler = new JwtSecurityTokenHandler();
        var jwtToken = handler.ReadJwtToken(bearerToken);
        
        // Extract CHash claim (contains Base64-encoded AES key)
        var chashClaim = jwtToken.Claims.FirstOrDefault(c => c.Type == "chash");
        if (chashClaim == null)
            throw new Exception("AES key not found in token");
        
        // Decode Base64 to get 32-byte key
        return Convert.FromBase64String(chashClaim.Value);
    }
    
    /// <summary>
    /// Encrypts a JSON string using AES-256-CBC
    /// </summary>
    public static string EncryptPayload(string plainText, byte[] key)
    {
        using var aes = Aes.Create();
        aes.Key = key;
        aes.IV = IV;
        aes.Mode = CipherMode.CBC;
        aes.Padding = PaddingMode.PKCS7;
        
        using var encryptor = aes.CreateEncryptor();
        using var ms = new MemoryStream();
        using (var cs = new CryptoStream(ms, encryptor, CryptoStreamMode.Write))
        using (var sw = new StreamWriter(cs, Encoding.UTF8))
        {
            sw.Write(plainText);
        }
        
        return Convert.ToBase64String(ms.ToArray());
    }
    
    /// <summary>
    /// Decrypts a Base64-encoded encrypted string
    /// </summary>
    public static string DecryptPayload(string encryptedBase64, byte[] key)
    {
        var encryptedBytes = Convert.FromBase64String(encryptedBase64);
        
        using var aes = Aes.Create();
        aes.Key = key;
        aes.IV = IV;
        aes.Mode = CipherMode.CBC;
        aes.Padding = PaddingMode.PKCS7;
        
        using var decryptor = aes.CreateDecryptor();
        using var ms = new MemoryStream(encryptedBytes);
        using var cs = new CryptoStream(ms, decryptor, CryptoStreamMode.Read);
        using var sr = new StreamReader(cs, Encoding.UTF8);
        
        return sr.ReadToEnd();
    }
}

// Usage Example
public class FundsTransferClient
{
    private string _bearerToken;
    private byte[] _aesKey;
    
    public async Task<string> AuthenticateAsync(string clientId, string clientSecret, int institutionId)
    {
        // Step 1: Get bearer token
        var tokenResponse = await GetBearerTokenAsync(clientId, clientSecret, institutionId);
        _bearerToken = tokenResponse.bearerToken;
        
        // Step 2: Extract AES key from token
        _aesKey = EncryptionHelper.ExtractAesKeyFromToken(_bearerToken);
        
        return _bearerToken;
    }
    
    public async Task<FundsTransferResponse> TransferFundsAsync(FundsTransferRequest request)
    {
        // Step 3: Serialize request to JSON
        var jsonRequest = JsonSerializer.Serialize(request);
        
        // Step 4: Encrypt the JSON
        var encryptedData = EncryptionHelper.EncryptPayload(jsonRequest, _aesKey);
        
        // Step 5: Create encrypted payload
        var encryptedPayload = new { data = encryptedData };
        
        // Step 6: Send request
        var response = await HttpClient.PostAsync(
            "/api/v1/xapi-service/fundsTransfer",
            new StringContent(JsonSerializer.Serialize(encryptedPayload), 
                Encoding.UTF8, "application/json"),
            headers: new Dictionary<string, string>
            {
                { "Authorization", $"Bearer {_bearerToken}" }
            });
        
        var responseJson = await response.Content.ReadAsStringAsync();
        var encryptedResponse = JsonSerializer.Deserialize<EncryptedPayload>(responseJson);
        
        // Step 7: Decrypt response
        var decryptedJson = EncryptionHelper.DecryptPayload(encryptedResponse.data, _aesKey);
        return JsonSerializer.Deserialize<FundsTransferResponse>(decryptedJson);
    }
}

4. Extract AES Key from JWT Token

The JWT bearer token contains the AES encryption key in the Chash claim. You need to:

  1. Decode the JWT token (without verification for key extraction)
  2. Extract the Chash claim value
  3. Decode the Base64 string to get the 32-byte AES key
Note: The CHash claim contains a Base64-encoded 256-bit AES key. This key is unique for each bearer token and should be used for all encrypted requests during the token's validity period.

5. Prepare Request Payload

Create your JSON request object as you would for a non-encrypted endpoint. The following example shows a typical structure for a fund transfer request; you will encrypt this entire payload in the next step.

Example (fund transfer)

{
  "institutionId": 9,
  "channelId": 5,
  "data": [
    {
      "drAcctNo": "1011000001",
      "drAcctType": "SAV",
      "crAcctNo": "1011000002",
      "crAcctType": "SAV",
      "amount": 10000.00,
      "crncyCode": "UGX",
      "tranCode": "FT",
      "description": "Fund Transfer",
      "transactionReference": "REF-001"
    }
  ]
}

6. Encrypt the Payload

Encrypt the JSON string (e.g. the payload from Prepare Request Payload) using AES-256-CBC with the extracted key:

  1. Convert the JSON payload to UTF-8 bytes.
  2. Use AES-256-CBC encryption with PKCS7 padding.
  3. Use a fixed Initialization Vector (IV): 1234567890123456 (16 bytes).
  4. Encode the encrypted bytes to Base64.

7. Send Encrypted Request

Wrap the encrypted Base64 string in an EncryptedPayload object and send it with the required headers.

Payload format: A JSON object with a single key data whose value is the Base64-encoded encrypted payload.

Request headers

HeaderValue
AuthorizationBearer {your-bearer-token}
Content-Typeapplication/json

Request body example

{
  "data": "U2FsdGVkX1+3x4y5z6a7b8c9d0e1f2g3h415j6k718m9n0o1p2q3r4s5t6u7v8w9x0y1z2..."
}

8. Decrypt the Response

The server response is also encrypted. The response body is a JSON object with a data field containing a Base64-encoded string. Decode from Base64, then decrypt using the same AES-256-CBC key and IV (1234567890123456) to obtain the plain JSON.

Encrypted response body example

{
  "data": "U2FsdGVkX1+9x8y7z6a5b4c3d2e1f0g9h817j6k514m3n2o1p0q9r8s7t6u5v4w3x2y1z0..."
}

Decrypt the data field using the same AES key and IV to get the JSON response:

Decrypted response example

{
  "responseCode": "0",
  "responseMessage": "Transfer successful",
  "data": {
    "transactionId": 12345
  }
}

9. Response Codes

Application-level response codes used by the Authentication API. These are aligned with the Card Integration API response codes.

CodeDescription
0Success
400Bad Request — Invalid or malformed request payload
401Unauthorized — Missing or invalid authentication
403Forbidden — Access not allowed for this resource
404Not Found — Resource or endpoint not found
422Unprocessable Entity — Validation error (e.g. invalid input)
500Internal Server Error — Server-side error; retry or contact support
-98Validation error — Required field missing or invalid
-99Business / general error — See responseMessage for details
